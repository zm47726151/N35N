package Oct2017.day0724;

public class Ques0724 {
	class Dog {

		String name;

		public Dog(String name) {
			super();
			this.name = name;
		}

		public String toString() {
			return "name:" + name;
		}
	}

	class BigDog extends Dog {

		String color;

		public BigDog(String name, String color) {
			this.name = name;
			this.color = color;
		}

		public String toString() {
			return "name:" + name + ", color:" + color;
		}
	}

}
